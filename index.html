*Will work

// Called upon receiving a Node/Express HTTP request.
co(function *() {
  var products = yield loadProductsFromDb;
  res.send(products);
})();

By combining 1) a particular generator consumer with 2) a particular generator function, we get asynchronous code that looks remarkably like synchronous code.

  var fn1 = 1;

    var current = fn2;
    fn2 = fn1;
    fn1 = fn1 + current;

Toy Example 1

function *kaijuPredictor() {
  var interval = 169; // Days between first and second occurence.
  while (true) {
    console.log('The next Kaiju will appear in %d days.', interval);
    yield null;
    interval /= 2;
  }
}

Toy Example 2

function *voicemail() {
  var messages = loadTheMessages(); // Returns an array of strings. Assume it's synchronous.
  
  var index = 0, response = 0;
  while (index < messages.length) {
    console.log('Press 0 to repeat. Press 1 to go to next message.');
    response = yield messages[index]; // <-- THE IMPORTANT LINE.
    if (response === 1) index++;
  }
}

Improvised Generator Consumer

function what(genFunc) {
  var generator = genFunc();
  var weAreHere = generator.next();
  bam(weAreHere, generator);
}

function bam(weAreHere, generator) {
  if (weAreHere.done === false) {
    var asyncFunction = weAreHere.value;
    asyncFunction(function (err, data) {
      if (err) throw err;
      weAreHere = generator.next(data);
      bam(weAreHere, generator);
    });
  }
}

module.exports = what;
 
var fs = require('fs');
var what = require('what');

what(function *() {
  var a = yield thunkReadFile('./cat.txt');
  console.log(a);
  var b = yield thunkReadFile('./dog.txt');;
  console.log(b);
  var a = yield thunkReadFile('./cat.txt');
  console.log(a);
});

function thunkReadFile(path) {
  return function (callback) {
    fs.readFile(path, {encoding: 'utf8'}, callback)
  };
}
